const CACHE_NAME_MATCHERS = ["daekwang", "app-shell", "runtime"];
const LEGACY_CACHE_NAMES = ["daekwang-app-shell-v1", "daekwang-runtime-v1"];

function shouldDeleteCache(cacheName) {
  return LEGACY_CACHE_NAMES.includes(cacheName) || CACHE_NAME_MATCHERS.some((matcher) => cacheName.includes(matcher));
}

async function clearDaekwangCaches() {
  const keys = await caches.keys();
  await Promise.all(keys.filter(shouldDeleteCache).map((key) => caches.delete(key)));
}

async function notifyClients() {
  const clients = await self.clients.matchAll({ includeUncontrolled: true, type: "window" });
  clients.forEach((client) => client.postMessage({ type: "SW_CACHE_CLEARED" }));
}

self.addEventListener("install", (event) => {
  event.waitUntil(self.skipWaiting());
});

self.addEventListener("activate", (event) => {
  event.waitUntil(
    clearDaekwangCaches()
      .then(() => self.clients.claim())
      .then(() => notifyClients()),
  );
});

self.addEventListener("fetch", (event) => {
  const { request } = event;
  if (request.method !== "GET") return;

  const url = new URL(request.url);
  if (url.origin !== self.location.origin) return;

  event.respondWith(fetch(request));
});
